```python
def longesr_words(file):
    with open(file, encoding='utf-8') as f:
        words = f.read().split()
        max_length = len(max(words, key=len))
        for word in words:
            if len(word) == max_length:
                sought_words = word
        if len(sought_words) == 1:
            return sought_words[0]
        return sought_words
print(longesr_words('input.txt'))
```

```python
f = open('text.txt', 'r', encoding="utf-8")
digits = [".", ",", ":"]


def remove_digits(word):
    result = ""
    for digit in digits:
        result = word.replace(digit, "")
    return result


strs = f.readlines()
d = dict()
for str in strs:
    for word in str.split(" "):
        w = word.rstrip()
        w = remove_digits(w)
        if w in d.keys():
            d[w] += 1
        else:
            d[w] = 1
t = []
for key in d.keys():
    t.append((key, d[key]))
t.sort(key=lambda x: x[1], reverse=True)
f.close()

print(d)
print(t[0])
```
