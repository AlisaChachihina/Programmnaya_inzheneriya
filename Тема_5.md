def superset(set_1, set_2):
    if set_1 > set_2:
        print(f'Объект {set_1} является чистым супермножеством')
    elif set_1 == set_2:
        print('Множества равны')
    elif set_1 < set_2:
        print(f'Объект {set_2} является чистым супермножеством ')

if __name__ == '__main__':

### Преподаватель по математике придумал странную задачку. ```python
import math
one = [12, 25, 3, 48, 71]
two = [5, 18, 40, 62, 98]
three = [4, 21, 37, 56, 84]
a1 = max(one) #максимальный элемент первого списка
a2 = min(one) #минимальный элемент первого списка
b1 = max(two) #максимальный элемент второго списка
b2 = min(two) #минимальный элемент второго списка
c1 = max(three) #максимальный элемент третьего списка
c2 = min(three) #минимальный элемент третьего списка
p1 = (a1 + b1 + c1)/2 #полупериметр первого треугольника
p2 = (a2 + b2 + c2)/2 #полупериметр второго треугольника
s1 = math.sqrt(p1*(p1-a1)*(p1-b1)*(p1-c1)) #площадь первого треугольника
s2 = math.sqrt(p2*(p2-a2)*(p2-b2)*(p2-c2)) #площадь второго треугольника
print('Площадь первого трекгольника: ', s1)
print('Площадь второго треугольника: ', s2)

 ## Самостоятельная работа №5
### Вам предоставлены списки натуральных чисел, из них необходимо сформировать множества. При этом следует соблюдать это правило: если какое-либо число повторяется, то преобразовать его в строку по следующему образцу: например, если число 4 повторяется 3 раза, то в множестве будет следующая запись: само число 4, строка «44», строка «444».

```python
def mno(list): #создаем функцию
    a = set() #создаем пустое множество
    set_list = set(list) #преобразуем список в множество
    for i in set_list: #идем циклом по множеству set_list
        count = list.count(i) #создаем переменную, в которой считаем количество вхождений каждого элемента из множества set_list
        a.add(i) #добавляем число в множество а
        for l in range(1,count): #идем циклом от одного до количества вхождений
            a.add(f"{i}"*(l+1)) #добавляем в множество а строку равную элементу, записанному l раз
    return a #возвращаем получившееся множество

list_1 = [1, 2, 3, 4, 6] #первый список
list_2 = [5, 5, 5, 5, 5] #второй список
list_3 = [2, 2, 1, 2, 2, 5, 6, 7, 1, 3, 2, 2] #третий список

print(mno(list_1)) #выводим преобразованный первый список во множество
print(mno(list_2)) #выводим преобразованный второй список во множество
print(mno(list_3)) #выводим преобразованный третий список во множество

```

## Выводы
В данном коде мы создаем функцию mno, в которой получаем множество уникальных значений заданного списка, с помощью преобразования списка в множество, затем вычисляем количество их вхождений в изначательный список с помощью count, а в итоговое множество записываем само уникальное значение, а также строки вида f"{i}"*p, где i - уникальное значение,  2 <=р <=количеству вхождений.
